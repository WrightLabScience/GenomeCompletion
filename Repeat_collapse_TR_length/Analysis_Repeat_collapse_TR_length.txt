### Balanced genome dataset for species_taxid and read_type
# list of complete genome, single contig, and no ambiguouty
refseq_bacteria_assembly_summary_04112026_complete <- readRDS('Collecting_genomes_and_metadata/refseq_bacteria_assembly_summary_04112026_complete.rds')

# check genome counts
library(dplyr)
species_counts <- refseq_bacteria_assembly_summary_04112026_complete %>%
    group_by(species_taxid, read_type) %>%
    summarise(n = n(), .groups = "drop") %>%
    tidyr::pivot_wider(names_from = read_type, values_from = n, values_fill = 0) %>%
    filter(s >= 1, h >= 1, l >= 1)

# get min_n per species
species_min_n <- species_counts %>%
    mutate(min_n = pmin(s, h, l)) %>%
    select(species_taxid, min_n)

# sample equal numbers per species per technology
assembly_summary_balanced <- refseq_bacteria_assembly_summary_04112026_complete %>%
    filter(read_type %in% c("s", "h", "l")) %>%
    filter(species_taxid %in% species_counts$species_taxid) %>%
    left_join(species_min_n, by = "species_taxid") %>%
    group_by(species_taxid, read_type) %>%
    group_modify(~ slice_sample(.x, n = unique(.x$min_n))) %>%
    ungroup()

## K-S test, directional
pairs_mat <- combn(names(col_map_SeqTech), 2)

ks_results_2D <- apply(pairs_mat, 2, function(pair) {
    x <- l_TR_lens[[pair[1]]]
    y <- l_TR_lens[[pair[2]]]
    
    ks_two   <- ks.test(x, y, alternative = "two.sided")
    ks_less  <- ks.test(x, y, alternative = "less")
    ks_great <- ks.test(x, y, alternative = "greater")
    
    data.frame(group1  = pair[1], group2 = pair[2],
               D       = ks_two$statistic,
               p_two   = ks_two$p.value,
               p_less  = ks_less$p.value,
               p_great = ks_great$p.value
               )
})

ks_results_2D_df <- do.call(rbind, ks_results_2D)
ks_results_2D_df$p_two_adj   <- p.adjust(ks_results_2D_df$p_two,   method = "BH")
ks_results_2D_df$p_less_adj  <- p.adjust(ks_results_2D_df$p_less,  method = "BH")
ks_results_2D_df$p_great_adj <- p.adjust(ks_results_2D_df$p_great, method = "BH")
# direction based on p adj
ks_results_2D_df$direction <- ifelse(
    ks_results_2D_df$p_less_adj  > ks_results_2D_df$p_great_adj,
    paste0(ks_results_2D_df$group1, "<", ks_results_2D_df$group2),
    paste0(ks_results_2D_df$group1, ">", ks_results_2D_df$group2)
)
